import { FieldSchema } from '@lowcode/types';

export function parseSchema(rawSchema: any): FieldSchema[] {
  const children = rawSchema?.componentsTree?.[0]?.children ?? [];
  return children.map((child: any) => {
    const {
      id, label, field, required,
      showInList, showInDetail,
      ...rest
    } = child.props ?? {};

    return {
      id,
      label,
      field,
      type: child.componentName,
      required:     required     ?? false,
      showInList:   showInList   ?? true,
      showInDetail: showInDetail ?? true,
      props: rest,
    };
  });
}

export class FormSchemaEntity {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column({ unique: true })
  formUuid: string;

  @Column()
  name: string;

  @Column()
  modelName: string;

  @Column({ type: 'jsonb' })
  schemaJson: any;

  @CreateDateColumn()
  createdAt: Date;

  @UpdateDateColumn()
  updatedAt: Date;

  @BeforeInsert()
  generateFormUuid() {
    if (!this.formUuid) this.formUuid = uuidv4();
  }
}

3. SchemaService 实现：

save(dto: SaveSchemaDto):
- 有 dto.formUuid → findOne 找到记录更新后 save
- 无 dto.formUuid → 新建 entity 后 save
- 保存成功后调用 createTableFromSchema(modelName, schemaJson)
- 返回保存后的记录

findByUuid(formUuid):
- findOne({ where: { formUuid } })
- 找不到 throw new NotFoundException('表单不存在')

findAll():
- find({ order: { createdAt: 'DESC' } })

deleteByUuid(formUuid):
- 先 findByUuid 确认存在
- 再 delete

createTableFromSchema(modelName, schemaJson):
- 调用 parseSchema(schemaJson) 解析 fields
- 用 fieldTypeToDbType 映射类型
- 获取 QueryRunner：
  const qr = this.dataSource.createQueryRunner();
  await qr.connect();
- 执行建表 SQL：
  CREATE TABLE IF NOT EXISTS "{modelName}" (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    created_at TIMESTAMP DEFAULT now(),
    updated_at TIMESTAMP DEFAULT now()
  )
- 对每个 field 执行：
  ALTER TABLE "{modelName}"
  ADD COLUMN IF NOT EXISTS "{field.field}" {dbType} NULL
- 释放：await qr.release()
- 包裹 try/catch，catch 中 await qr.release() 后 throw

    );
  }
}

6. 创建 src/common/http-exception.filter.ts：
@Catch()
export class HttpExceptionFilter implements ExceptionFilter {
  catch(exception, host) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();
    const status = exception instanceof HttpException
      ? exception.getStatus()
      : 500;
    response.status(status).json({
      code: -1,
      data: null,
      message: exception.message ?? '服务器错误',
    });
  }
}

  fields = Object.keys(dto.data)

updateRecord(dto: UpdateRecordDto):
- modelName = await getModelName(dto.formUuid)

  SET {field=$1, ...}, updated_at=now()

  RETURNING *

deleteRecord(dto: DeleteRecordDto):

Task 15 — 实现 schemaToColumns 工具函数
1. 创建 src/utils/schemaToColumns.ts

2. 实现 schemaToColumns(fields: FieldSchema[]): ProFormColumnsType[]

每个 field 映射规则，基础结构：
{
  title:     field.label,
  dataIndex: field.field,
  formItemProps: {
    rules: field.required
      ? [{ required: true, message: `请输入${field.label}` }]
      : [],
  },
}

根据 field.type 追加 valueType 和 fieldProps：

ProFormText:
  valueType: 'text'

ProFormTextArea:
  valueType: 'textarea'
  fieldProps: { rows: field.props?.rows ?? 4 }

ProFormDigit:
  valueType: 'digit'
  fieldProps: {
    min:       field.props?.min ?? 0,
    max:       field.props?.max ?? 999999,
    precision: field.props?.precision ?? 0,
  }

ProFormSelect:
  valueType: 'select'
  valueEnum: field.props?.options?.reduce((acc, opt) => {
    acc[opt.value] = { text: opt.label };
    return acc;
  }, {}) ?? {}

ProFormDatePicker:
  valueType: 'date'
  fieldProps: { format: field.props?.format ?? 'YYYY-MM-DD' }

ProFormSwitch:
  valueType: 'switch'

ProFormRadio:
  valueType: 'radio'
  valueEnum: 同 ProFormSelect 转换方式

ProFormCheckbox:
  valueType: 'checkbox'
  valueEnum: 同 ProFormSelect 转换方式
